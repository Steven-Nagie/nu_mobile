** SCHEMA **
CREATE TABLE users
(
	id SERIAL PRIMARY KEY,
  firstname TEXT,
  lastname TEXT,
  state VARCHAR(2),
  email TEXT UNIQUE,
  password TEXT,
  title TEXT, 
	photo TEXT
)

CREATE TABLE scores
(
	id SERIAL PRIMARY KEY,
  total INTEGER,
	transport INTEGER,
	energy INTEGER,
	water INTEGER,
	food INTEGER,
	waste INTEGER,
	userId INTEGER
		REFERENCES users(id)
)
